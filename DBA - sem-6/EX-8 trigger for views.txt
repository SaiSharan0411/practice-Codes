CREATE TABLE EMPLOYEE(EMPNO NUMBER,NAME VARCHAR(20),JOB VARCHAR(20));

CREATE TABLE SALARY(EMPNO NUMBER,BASIC NUMBER,HRA NUMBER);

CREATE VIEW V1 AS SELECT E.EMPNO,NAME,JOB,BASIC,HRA FROM EMPLOYEE E,SALARY S WHERE E.EMPNO=S.EMPNO;

CREATE TRIGGER TR_VIEW INSTEAD OF INSERT ON V1
FOR EACH ROW
BEGIN
INSERT INTO EMPLOYEE(EMPNO,NAME,JOB)VALUES(:NEW.EMPNO,:NEW.NAME,:NEW.JOB);
INSERT INTO SALARY(BASIC,HRA)VALUES(:NEW.BASIC,:NEW.HRA);
END;
