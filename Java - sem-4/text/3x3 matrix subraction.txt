import java.io.*;
class matrixsub
{
public static void main(String[]args)
{
int x[][]={{2,3,4},{5,2,3},{2,3,5}};
int y[][]={{4,5,3},{5,6,3},{4,5,4}};
int z[][]=new int[3][3];
for(int i=0;i<3;i++)
{
for(int j=0;j<3;j++)
{
z[i][j]=x[i][j]-y[i][j];
System.out.print(" "+z[i][j]+" ");
}
System.out.println();
}
}
}
