#include <stdio.h>
void main()
{
    int n, i;
    printf("Enter any number : ");
    scanf("%d", &n);
    if(n == 1)
    {
        printf("Neither Prime nor Composite.");
        exit(0);
    }
    else if (n <= 1)
    {
        printf("Not a prime number.");
        exit(0);
    }
    for(i = 2; i < n; i++)
    {
        if(n % i == 0)
        {
            printf("Not a prime number.");
            break;
        }
    }
    if(i == n)
      printf("Prime number.");
}
