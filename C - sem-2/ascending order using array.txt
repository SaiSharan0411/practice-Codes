#include <stdio.h>
 void main()
{
    int a[5], n , i, j, t;
    printf("Enter the size of the array : ");
    scanf("%d", &n);
    printf("\nEnter the array elements : \n");
    for(i = 0; i < n; i++)
      scanf("%d", &a[i]);
    for(i = 0; i < n-1; i++)
    {
        for(j = i + 1; j < n; j++)
        {
            if(a[i] > a[j])
            {
                t = a[i];
                a[i] = a[j];
                a[j] = t;
            }
        }
    }
    printf("\nThe sorted array is : \n");
    for(i = 0; i < n; i++)
      printf("\t%d\n", a[i]);
}
