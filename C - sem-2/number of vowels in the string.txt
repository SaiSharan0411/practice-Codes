#include <stdio.h>
#include <string.h>
void main() 
{
    char str[40];
    char list[] = "aeiouAEIOU";
    printf("Enter a string:");
    scanf("%s",str);
    int vow = 0;
    for(int i = 0; i<strlen(str); i++) {
        for(int j = 0; j<strlen(list); j++) {
            if(str[i] == list[j]){
                vow++;
                break;
        }
      }

    }
    printf("The string had %d vowels!",vow);
}
